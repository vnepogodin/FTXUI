48  for (size_t i = 0; i < entries_.size(); ++i) {
49  bool is_focused = (focused_entry() == int(i)) && is_menu_focused;

53  : option_->style_selected)

70  if (!CaptureMouse(event))

74  return OnMouseEvent(event);
